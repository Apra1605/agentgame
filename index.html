} }); // âœ… Start Express const PORT = process.env.PORT || 3000; app.listen(PORT, () => console.log(`ðŸŒ API server running on http://localhost:${PORT}`)); // âœ… Discord bot const bot = new Client({ intents: [ GatewayIntentBits.Guilds, GatewayIntentBits.GuildMessages, GatewayIntentBits.MessageContent, GatewayIntentBits.GuildMembers, GatewayIntentBits.GuildVoiceStates, GatewayIntentBits.GuildMessageReactions, GatewayIntentBits.DirectMessages, GatewayIntentBits.GuildPresences ], partials: [Partials.Message, Partials.Channel, Partials.Reaction] });
